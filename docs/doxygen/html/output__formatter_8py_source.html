141def print_vulnerability(cve_id, description, cve_url=None, score=None):
142 """
143 Print vulnerability information with clickable CVE.org URL.
144
145 @param cve_id str CVE identifier (e.g., "CVE-2024-1234")
146 @param description str Vulnerability description text
147 @param cve_url str (optional) URL to CVE details page
148 @param score float (optional) CVSS score if available
149
150 @details
151 Format example:
152 @code
153 [CVE-2024-1234] Vulnerability description text
154 ðŸ”— https://cve.mitre.org/cgi-bin/cvename.cgi?name=CVE-2024-1234
155 @endcode
156
157 URLs are formatted as OSC 8 hyperlinks for terminal click-through support
158 in compatible terminals (iTerm2, GNOME Terminal, Windows Terminal, etc.)
159 """
160 if cve_url:
161 # ANSI hyperlink format for terminal clickable links (OSC 8)
162 clickable_url = f"\033]8;;{cve_url}\033\\{cve_url}\033]8;;\033\\"
163 if score:
164 print(f" {Colors.RED}[{cve_id}]{Colors.RESET} {description}")
165 print(f" {Colors.BLUE}ðŸ”— {clickable_url}{Colors.RESET} (Score: {score})")
166 else:
167 print(f" {Colors.RED}[{cve_id}]{Colors.RESET} {description}")
168 print(f" {Colors.BLUE}ðŸ”— {clickable_url}{Colors.RESET}")
169 else:
170 if score:
171 print(f" {Colors.RED}[{cve_id}]{Colors.RESET} {description} (Score: {score})")
172 else:
173 print(f" {Colors.RED}[{cve_id}]{Colors.RESET} {description}")
