63def flush_all_caches():
64 """
65 Remove all cached data including package caches, CPE caches, and vulnerability database.
66
67 @details
68 Flushes:
69 - machines/cpe_cache.json (global CPE cache)
70 - machines/{machine}/installed_packages.json (per-machine package lists)
71 - machines/{machine}/vulnerability_report.json (generated reports)
72 - db/vulnerability_cache.db (SQLite vulnerability cache)
73 """
74 items_flushed = []
75
76 # Remove global CPE cache
77 cpe_cache = os.path.join(CACHE_DIR, "cpe_cache.json")
78 if os.path.exists(cpe_cache):
79 os.remove(cpe_cache)
80 items_flushed.append("Global CPE cache")
81 logger.info(f"Flushed: {cpe_cache}")
82
83 # Remove per-machine caches
84 machines_dir = os.path.join(CACHE_DIR, "machines")
85 if os.path.exists(machines_dir):
86 for machine_dir in os.listdir(machines_dir):
87 machine_path = os.path.join(machines_dir, machine_dir)
88 if os.path.isdir(machine_path):
89 # Remove installed_packages.json
90 pkg_cache = os.path.join(machine_path, "installed_packages.json")
91 if os.path.exists(pkg_cache):
92 os.remove(pkg_cache)
93 logger.info(f"Flushed: {pkg_cache}")
94
95 # Remove vulnerability_report.json
96 report = os.path.join(machine_path, "vulnerability_report.json")
97 if os.path.exists(report):
98 os.remove(report)
99 logger.info(f"Flushed: {report}")
100
101 if os.path.exists(pkg_cache) or os.path.exists(report):
102 items_flushed.append(f"Machine cache: {machine_dir}")
103
104 # Remove SQLite vulnerability database
105 db_file = os.path.join(CACHE_DIR, "vulnerability_cache.db")
106 if os.path.exists(db_file):
107 os.remove(db_file)
108 items_flushed.append("SQLite vulnerability cache")
109 logger.info(f"Flushed: {db_file}")
110
111 if items_flushed:
112 fmt.print_success(f"Cache flush completed. Removed: {', '.join(items_flushed)}")
113 logger.info(f"Cache flush completed. Removed: {items_flushed}")
114 else:
115 fmt.print_info("No caches found to flush")
116 logger.info("No caches found to flush")

119def parse_arguments():
120 """
121 Parse command-line arguments for inventory file, cache flushing, and forced checks.
122
123 @return argparse.Namespace Parsed arguments with attributes:
124 - inventory: Custom inventory file path
125 - flush_cache: Whether to flush all caches before scanning
126 - force_check: Whether to check all packages (not just new ones)
127 - report_only: Whether to only generate report from existing cache
128
129 @details
130 Supported arguments:
131 - --inventory FILE: Use custom inventory file (default: inventory.ini)
132 - --flush-cache: Remove all cached data before scanning
133 - --force-check: Check all packages even if no new ones detected (ignore cache)
134 - --report-only: Generate HTML report from existing cache data without scanning
135
136 Examples:
137 - python main.py --inventory custom_inventory.ini
138 - python main.py --flush-cache
139 - python main.py --force-check
140 - python main.py --report-only
141 - python main.py --inventory custom.ini --flush-cache --force-check
142 """
143 parser = argparse.ArgumentParser(
144 description="Vulnerability scanner for Linux machines with NVD integration"
145 )
146 parser.add_argument(
147 "--inventory",
148 type=str,
149 default=DEFAULT_INVENTORY,
150 help=f"Path to inventory file (default: {DEFAULT_INVENTORY})"
151 )
152 parser.add_argument(
153 "--flush-cache",
154 action="store_true",
155 help="Flush all caches (CPE cache, package cache, vulnerability database) before scanning"
156 )
157 parser.add_argument(
158 "--force-check",
159 action="store_true",
160 help="Check all installed packages for vulnerabilities, even if no new packages are detected"
161 )
162 parser.add_argument(
163 "--report-only",
164 action="store_true",
165 help="Generate HTML report from existing vulnerability cache data without scanning machines"
166 )
167 return parser.parse_args()
