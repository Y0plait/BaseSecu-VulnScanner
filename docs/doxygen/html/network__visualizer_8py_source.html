80def generate_network_svg_for_host(host_address, machine_name):
81 """
82 Generate SVG network topology diagram for a specific host using visualnet-scanner.
83
84 Executes the visualnet-scanner.sh script to perform an Nmap scan on the target
85 host and generate an SVG visualization of the network topology. Results are
86 cached for future report generation.
87
88 @param host_address str IP address or hostname to scan (e.g., "192.168.10.66")
89 @param machine_name str Machine identifier for caching (e.g., "srv01")
90
91 @return str Path to generated SVG file (cache/machines/{machine}/network_topology.svg)
92 or None if generation failed
93
94 @details
95 **Execution Flow:**
96
97 1. Verify visualnet-scanner.sh exists and is executable
98 2. Call: ./visualnet-scanner.sh {host_address} network_topology.svg
99 3. Script internally:
100 - Executes nmap scan (-T4 -F mode for speed)
101 - Converts Nmap XML to Graphviz DOT format via nmap-formatter
102 - Generates SVG using dot -Tsvg
103 4. Capture SVG output from stdout
104 5. Save to cache/machines/{machine}/network_topology.svg
105
106 **Output Location:**
107
108 - File: cache/machines/{machine_name}/network_topology.svg
109 - Permissions: User readable
110 - Size: Typically 5-50KB depending on network size
111
112 **External Dependencies:**
113
114 - visualnet-scanner.sh: Main orchestration script (must be in project root)
115 - nmap: Network scanning tool (system package)
116 - nmap-formatter: Custom binary (project root)
117 - graphviz (dot): SVG generation (system package)
118
119 **Error Handling:**
120
121 - Missing script: Logs warning, returns None
122 - Non-executable script: Logs warning, returns None
123 - Nmap timeout: Falls back to informational SVG
124 - Format conversion failure: Falls back to informational SVG
125 - Empty output: Creates fallback SVG with machine info
126
127 **Timeouts:**
128
129 - Default: 5 minutes (300 seconds)
130 - Covers: Nmap scan + format conversion + SVG generation
131
132 @note
133 - Logs execution details to logger for audit trail
134 - Uses subprocess.run() with capture_output=True, text=True
135 - Suppresses stderr to avoid log pollution
136 - Creates fallback SVG if primary scanning fails
137 """
138 try:
139 # Determine the scan target - use the host address
140 scan_target = host_address
141
142 # Create output directory for this machine
143 machine_dir = os.path.join(CACHE_DIR, "machines", machine_name)
144 os.makedirs(machine_dir, exist_ok=True)
145
146 # Define output SVG file path
147 svg_output = os.path.join(machine_dir, "network_topology.svg")
148
149 # Get the project root directory (go up from src/reporting/ to project root)
150 # __file__ â†’ /path/src/reporting/network_visualizer.py
151 # dirname â†’ /path/src/reporting
152 # dirname â†’ /path/src
153 # dirname â†’ /path (project root)
154 current_file = os.path.abspath(__file__) # Full path to this file
155 src_dir = os.path.dirname(current_file) # src/reporting/
156 src_parent = os.path.dirname(src_dir) # src/
157 project_root = os.path.dirname(src_parent) # project root
158
159 scanner_script = os.path.join(project_root, "visualnet-scanner.sh")
160
161 # Check if scanner script exists
162 if not os.path.exists(scanner_script):
163 logger.warning(f"visualnet-scanner.sh not found at {scanner_script}")
164 return None
165
166 # Check if script is executable
167 if not os.access(scanner_script, os.X_OK):
168 logger.warning(f"visualnet-scanner.sh is not executable: {scanner_script}")
169 return None
170
171 logger.info(f"Generating network SVG for {machine_name} ({scan_target})...")
172
173 # Execute the scanner script
174 # Syntax: ./visualnet-scanner.sh <target> <output_filename>
175 # The script outputs SVG to stdout and logs to stderr
176 result = subprocess.run(
177 [scanner_script, scan_target, "network_topology.svg"],
178 capture_output=True,
179 text=True,
180 timeout=300 # 5 minute timeout for network scan
181 )
182
183 # Check if execution was successful
184 if result.returncode == 0 and result.stdout:
185 # The script outputs the SVG content to stdout
186 svg_content = result.stdout
187 with open(svg_output, 'w') as f:
188 f.write(svg_content)
189 logger.info(f"âœ“ Network SVG generated for {machine_name}")
190 return svg_output
191 else:
192 logger.error(f"Network scan failed for {machine_name}")
193 if result.stderr:
194 logger.debug(f"Script stderr: {result.stderr}")
195 # Try to create a fallback informational SVG
196 logger.info(f"Creating fallback network information SVG for {machine_name}")
197 svg_fallback = create_fallback_network_svg(host_address, machine_name)
198 if svg_fallback:
199 with open(svg_output, 'w') as f:
200 f.write(svg_fallback)
201 logger.info(f"Created fallback SVG for {machine_name}")
202 return svg_output
203 return None
204
205 # Verify output file was created
206 if not os.path.exists(svg_output):
207 logger.error(f"SVG file was not created for {machine_name}")
208 # Try fallback
209 svg_fallback = create_fallback_network_svg(host_address, machine_name)
210 if svg_fallback:
211 with open(svg_output, 'w') as f:
212 f.write(svg_fallback)
213 return svg_output
214 return None
215
216 # Check if SVG file has content
217 file_size = os.path.getsize(svg_output)
218 if file_size == 0:
219 logger.error(f"Generated SVG file is empty for {machine_name}")
220 # Try fallback
221 svg_fallback = create_fallback_network_svg(host_address, machine_name)
222 if svg_fallback:
223 with open(svg_output, 'w') as f:
224 f.write(svg_fallback)
225 return svg_output
226 return None
227
228 logger.info(f"Successfully generated network SVG for {machine_name} ({file_size} bytes)")
229 return svg_output
230
231 except subprocess.TimeoutExpired:
232 logger.error(f"Network scan timed out for {machine_name}")
233 return None
234 except Exception as e:
235 logger.error(f"Error generating network SVG for {machine_name}: {e}")
236 return None

239def svg_to_base64(svg_file_path):
240 """
241 Convert SVG file to base64 encoding for embedding in HTML.
242
243 @param svg_file_path str Path to SVG file
244
245 @return str Base64 encoded SVG content, or None if file not found/readable
246
247 @details
248 Converts SVG file to base64 for direct embedding in HTML via data URI.
249 This allows the SVG to be included without external file dependencies.
250 """
251 try:
252 if not os.path.exists(svg_file_path):
253 logger.warning(f"SVG file not found: {svg_file_path}")
254 return None
255
256 with open(svg_file_path, 'rb') as f:
257 svg_content = f.read()
258
259 # Encode to base64
260 encoded = base64.b64encode(svg_content).decode('utf-8')
261 logger.debug(f"Converted SVG to base64: {svg_file_path}")
262 return encoded
263
264 except Exception as e:
265 logger.error(f"Error converting SVG to base64: {e}")
266 return None

297def generate_network_visualizations(machines_config):
298 """
299 Generate network visualizations for all machines in the inventory.
300
301 @param machines_config dict Machine configuration from inventory
302 {machine_name: {'host': ip, 'user': user, ...}}
303
304 @return dict Network visualization data
305 {machine_name: {
306 'svg_path': 'path/to/file.svg',
307 'svg_content': '<svg>...</svg>',
308 'generated': True/False
309 }}
310
311 @details
312 Iterates through all Linux machines in the inventory and generates
313 network topology SVGs for their respective networks.
314 Returns both file paths and inline SVG content for flexible inclusion in reports.
315 """
316 network_visualizations = {}
317
318 logger.info(f"Generating network visualizations for {len(machines_config)} machines...")
319 print(f"[*] Generating network topology diagrams...")
320
321 for machine_name, machine_config in machines_config.items():
322 if machine_config.get('type') != 'linux':
323 logger.debug(f"Skipping {machine_name}: not a Linux machine")
324 continue
325
326 host_address = machine_config.get('host')
327 if not host_address:
328 logger.warning(f"No host address found for {machine_name}")
329 continue
330
331 # Generate SVG for this host
332 svg_path = generate_network_svg_for_host(host_address, machine_name)
333
334 network_visualizations[machine_name] = {
335 'svg_path': svg_path,
336 'svg_content': None,
337 'generated': False
338 }
339
340 if svg_path:
341 # Read SVG content for embedding
342 svg_content = read_svg_content(svg_path)
343 network_visualizations[machine_name]['svg_content'] = svg_content
344 network_visualizations[machine_name]['generated'] = True
345 logger.info(f"Network visualization ready for {machine_name}")
346 print(f"[âœ“] Network diagram generated for {machine_name}")
347 else:
348 logger.warning(f"Failed to generate network visualization for {machine_name}")
349 print(f"[âš ] Could not generate diagram for {machine_name} (see logs for details)")
350
351 successful = sum(1 for v in network_visualizations.values() if v['generated'])
352 logger.info(f"Network visualization generation complete: {successful} successful")
353 return network_visualizations
