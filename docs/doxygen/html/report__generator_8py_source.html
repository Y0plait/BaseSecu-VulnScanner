98def save_machine_report(machine_name, vulnerabilities_data):
99 """
100 Generate and save JSON vulnerability report for a machine.
101
102 Creates a timestamped JSON file containing all vulnerabilities found
103 on a machine, organized by CPE (Common Platform Enumeration) with
104 clickable CVE references and publication dates.
105
106 @param machine_name str Name of the machine being reported
107 Format: hostname or identifier (e.g., "srv01")
108 @param vulnerabilities_data dict Vulnerability data structure
109 Format: {cpe: [{cve_id, description, published_date}, ...], ...}
110
111 @return str Path to generated report file, or None on failure
112 Example: "cache/machines/srv01/vulnerability_report.json"
113
114 @details
115 **Output File Location:**
116
117 - Directory: cache/machines/{machine_name}/
118 - Filename: vulnerability_report.json
119 - Permissions: User readable
120 - Format: JSON with 2-space indentation
121
122 **Report Structure:**
123
124 ```json
125 {
126 "machine": "srv01",
127 "timestamp": "2025-12-24T14:30:45.123456",
128 "cpes_with_vulnerabilities": [
129 {
130 "cpe": "cpe:2.3:a:vendor:product:version:*:*:*:*:*:*:*",
131 "vulnerabilities": [
132 {
133 "cve_id": "CVE-2024-1234",
134 "description": "Vulnerability description from NVD",
135 "cve_url": "https://cve.mitre.org/cgi-bin/cvename.cgi?name=CVE-2024-1234",
136 "published_date": "2024-05-15T00:00:00Z"
137 }
138 ]
139 }
140 ]
141 }
142 ```
143
144 **Data Processing:**
145
146 1. Creates machine directory if it doesn't exist
147 2. Iterates through vulnerabilities_data dictionary
148 3. For each CPE with vulnerabilities:
149 - Creates CPE entry with empty vulnerabilities list
150 - For each vulnerability:
151 - Extracts cve_id, description, published_date
152 - Generates CVE URL using generate_cve_url()
153 - Appends to vulnerabilities list
154 - Appends CPE entry to report
155 4. Writes report as formatted JSON
156 5. Logs success or failure
157
158 **Error Handling:**
159
160 - Missing machine name: Returns None
161 - Empty vulnerabilities_data: Creates empty report
162 - File write failure: Logs error, returns None
163 - Invalid UTF-8: Logs and skips problematic entries
164
165 **Timestamp Format:**
166
167 - ISO 8601 format with microseconds
168 - Timezone: Local system timezone
169 - Example: "2025-12-24T14:30:45.123456"
170
171 @note
172 - Report is human-readable (not minified)
173 - Directory is created automatically if missing
174 - Overwrites existing report without warning
175 - Logs generation details to logger for audit trail
176 """
177 report_dir = os.path.join(CACHE_DIR, "machines", machine_name)
178 if not os.path.exists(report_dir):
179 os.makedirs(report_dir)
180
181 # Build report structure
182 report = {
183 "machine": machine_name,
184 "timestamp": datetime.now().isoformat(),
185 "cpes_with_vulnerabilities": []
186 }
187
188 # Process vulnerabilities by CPE
189 for cpe, vulns in vulnerabilities_data.items():
190 if vulns: # Only include CPEs with vulnerabilities
191 cpe_entry = {
192 "cpe": cpe,
193 "vulnerabilities": []
194 }
195
196 for vuln in vulns:
197 cve_entry = {
198 "cve_id": vuln["cve_id"],
199 "description": vuln["description"],
200 "cve_url": generate_cve_url(vuln["cve_id"])
201 }
202 if "published_date" in vuln:
203 cve_entry["published_date"] = vuln["published_date"]
204 cpe_entry["vulnerabilities"].append(cve_entry)
205
206 report["cpes_with_vulnerabilities"].append(cpe_entry)
207
208 # Save report to JSON file
209 report_file = f"{report_dir}/vulnerability_report.json"
210 try:
211 with open(report_file, "w") as f:
212 f.write(json.dumps(report, indent=2))
213 logger.info(f"Vulnerability report saved for {machine_name}: {report_file}")
214 return report_file
215 except Exception as e:
216 logger.error(f"Error saving vulnerability report for {machine_name}: {e}")
217 return None

220def get_total_vulnerabilities_from_report(report_dict):
221 """
222 Count total vulnerabilities in a report structure.
223
224 @param report_dict dict Report dictionary (as generated by save_machine_report)
225
226 @return int Total number of vulnerabilities across all CPEs
227
228 @details
229 Useful for statistics and summary generation.
230 """
231 total = 0
232 for cpe_entry in report_dict.get("cpes_with_vulnerabilities", []):
233 total += len(cpe_entry.get("vulnerabilities", []))
234 return total
