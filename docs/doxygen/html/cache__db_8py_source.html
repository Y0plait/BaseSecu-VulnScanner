46def sync_modified_cves(api_key=None):
47 """
48 Update cached CVE entries with latest information from NVD.
49
50 @param api_key str NVD API key for higher rate limits (optional)
51
52 @details
53 Queries NVD for any CVEs modified in the last 24 hours and updates
54 local database entries. Useful for periodic maintenance to keep cache fresh.
55
56 Does not check all CPEs, only looks for globally modified CVEs.
57 This is less efficient than selective updates but works without CPE list.
58 """
59 db = get_db()
60 try:
61 start_date = datetime.now() - timedelta(days=1)
62 logger.info(f"Checking NVD for any changes since {start_date}...")
63
64 # Query NVD for any CVEs modified in the window
65 updates = nvdlib.searchCVE(lastModStartDate=start_date, lastModEndDate=datetime.now(), key=api_key)
66
67 for cve in updates:
68 # Check if this CVE is already in our 'vulnerabilities' table
69 # If it is, update its description or impact score
70 db.execute("UPDATE vulnerabilities SET description = ? WHERE cve_id = ?",
71 (cve.descriptions[0].value, cve.id))
72 db.commit()
73 logger.info(f"Synced {len(updates)} modified CVEs from NVD")
74 finally:
75 db.close()
76 logger.debug("Closed database connection for CVE sync")

79def get_db():
80 """
81 Initialize and return SQLite database connection with required schema.
82
83 @return sqlite3.Connection Database connection object with tables created
84
85 @details
86 Creates two tables if they don't exist:
87
88 **cpe_index table:**
89 - cpe_string TEXT PRIMARY KEY: The CPE to identify
90 - last_fetched TIMESTAMP: When this CPE was last queried from NVD
91 Used to track which CPEs have been searched and detect cache misses.
92
93 **vulnerabilities table:**
94 - cpe_string TEXT: Foreign key to cpe_index
95 - cve_id TEXT: CVE identifier (e.g., "CVE-2024-1234")
96 - description TEXT: Vulnerability description from NVD
97 - published_date TEXT: CVE publication date from NVD
98 Stores actual CVE data for cached CPEs.
99
100 Schema design allows:
101 - Fast lookup of cached CPEs (indexed on cpe_string)
102 - Multiple CVEs per CPE (one-to-many relationship)
103 - Quick retrieval of all vulnerabilities for a CPE
104
105 Database file location: db/vulnerability_cache.db
106 """
107 conn = sqlite3.connect(VULN_DB_PATH)
108 cursor = conn.cursor()
109 # Stores which CPEs we have searched
110 cursor.execute('''CREATE TABLE IF NOT EXISTS cpe_index
111 (cpe_string TEXT PRIMARY KEY, last_fetched TIMESTAMP)''')
112 # Stores the CVE IDs linked to those CPEs
113 cursor.execute('''CREATE TABLE IF NOT EXISTS vulnerabilities
114 (cpe_string TEXT, cve_id TEXT, description TEXT, published_date TEXT,
115 FOREIGN KEY(cpe_string) REFERENCES cpe_index(cpe_string))''')
116
117 # Migration: Add published_date column if it doesn't exist (for existing databases)
118 cursor.execute("PRAGMA table_info(vulnerabilities)")
119 columns = [column[1] for column in cursor.fetchall()]
120 if 'published_date' not in columns:
121 try:
122 cursor.execute("ALTER TABLE vulnerabilities ADD COLUMN published_date TEXT")
123 conn.commit()
124 logger.info("Migration: Added published_date column to vulnerabilities table")
125 except Exception as e:
126 logger.debug(f"Migration note: {e}")
127
128 conn.commit()
129 return conn

132def get_vulnerabilities(cpe_string, api_key=None):
133 """
134 Retrieve vulnerabilities for a CPE using cache-first strategy.
135
136 @param cpe_string str CPE identifier (e.g., "cpe:2.3:a:vendor:product:version:...")
137 @param api_key str NVD API key for higher rate limits (optional)
138
139 @return list List of (cve_id, description) tuples for the CPE
140
141 @details
142 **Cache-first lookup algorithm:**
143 1. Check if CPE exists in local cpe_index table
144 2. If found (cache hit):
145 - Retrieve and return cached vulnerability data
146 - Timestamp shows when data was last fetched
147 3. If not found (cache miss):
148 - Query NVD API via nvdlib.searchCVE()
149 - Insert CPE into cpe_index with current timestamp
150 - Insert all returned CVEs into vulnerabilities table
151 - Return the CVE data
152
153 This approach:
154 - Minimizes NVD API calls (crucial for rate limiting)
155 - Provides instant results for frequently checked packages
156 - Automatically caches new packages on first query
157 - Rate limiting is handled by vulnerability_checker.py (0.6s delay)
158
159 The nvdlib library handles:
160 - HTTP connection to NVD
161 - Parsing JSON responses
162 - Rate limit exceptions (429, 503, etc)
163
164 @note
165 Caller should implement rate limiting between calls to respect
166 50 requests/30 seconds NVD API limit (with key).
167 """
168 db = get_db()
169 cursor = db.cursor()
170
171 try:
172 # 1. Check local index
173 cursor.execute("SELECT last_fetched FROM cpe_index WHERE cpe_string = ?", (cpe_string,))
174 row = cursor.fetchone()
175
176 if row:
177 print(f"[*] Cache Hit for {cpe_string}. Fetching from local DB...")
178 try:
179 # Try to query with published_date column (new schema)
180 cursor.execute("SELECT cve_id, description, published_date FROM vulnerabilities WHERE cpe_string = ? ORDER BY published_date DESC", (cpe_string,))
181 results = cursor.fetchall()
182 except Exception as e:
183 # Fallback for old schema without published_date
184 if 'no such column' in str(e):
185 logger.debug(f"Old schema detected, querying without published_date: {e}")
186 cursor.execute("SELECT cve_id, description FROM vulnerabilities WHERE cpe_string = ?", (cpe_string,))
187 old_results = cursor.fetchall()
188 # Convert to 3-tuple format with None for published_date
189 results = [(cve_id, desc, None) for cve_id, desc in old_results]
190 else:
191 raise
192 # Logic: You'd return these to the user now,
193 # then optionally call a 'sync_updates()' function to refresh the index.
194 return results
195
196 else:
197 print(f"[!] Cache Miss for {cpe_string}. Querying NVD API...")
198 # 2. Search NVD via nvdlib
199 # Note: cveList contains objects with .id and .descriptions[0].value
200 cve_list = nvdlib.searchCVE(cpeName=cpe_string, key=api_key)
201
202 # 3. Add to local index (Upsert)
203 now = datetime.now().isoformat()
204 cursor.execute("INSERT INTO cpe_index (cpe_string, last_fetched) VALUES (?, ?)", (cpe_string, now))
205
206 extracted_data = []
207 for cve in cve_list:
208 desc = cve.descriptions[0].value if cve.descriptions else "No description"
209 # Extract published date from CVE object
210 published_date = cve.published if hasattr(cve, 'published') else None
211 cursor.execute("INSERT INTO vulnerabilities (cpe_string, cve_id, description, published_date) VALUES (?, ?, ?, ?)",
212 (cpe_string, cve.id, desc, published_date))
213 extracted_data.append((cve.id, desc, published_date))
214
215 db.commit()
216 return extracted_data
217 finally:
218 db.close()
219 logger.debug(f"Closed database connection for CPE query: {cpe_string}")
