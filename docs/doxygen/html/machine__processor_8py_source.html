28def process_machine_packages(config, machine):
29 """
30 Retrieve and process installed packages for a machine.
31
32 Connects to a machine, retrieves all installed packages, compares them
33 with previously cached packages, and identifies newly installed packages.
34
35 @param config ConfigParser object containing machine configuration
36 @param machine str Machine name from inventory
37
38 @return tuple (installed_packages, new_packages)
39 installed_packages list: All currently installed packages
40 new_packages list: Packages not in the cache
41 Returns ([], []) if machine type is not supported
42
43 @details
44 - Creates machine directory if it doesn't exist
45 - Uses pkg_finder.get_installed_packages_linux() for SSH operations
46 - Compares with cached data using pkg_finder.get_new_packages()
47 - Updates package cache JSON file
48 """
49 if config[machine]['type'] != 'linux':
50 return [], []
51
52 # Ensure machine directory exists
53 machine_dir = os.path.join(CACHE_DIR, "machines", machine)
54 if not os.path.exists(machine_dir):
55 os.makedirs(machine_dir)
56 logger.debug(f"Created machine directory: {machine_dir}")
57
58 fmt.print_info("Retrieving installed packages...")
59 logger.debug(f"Retrieving installed packages from {machine}")
60 installed_packages = pkg_finder.get_installed_packages_linux(config, machine)
61 logger.info(f"Retrieved {len(installed_packages)} installed packages from {machine}")
62 fmt.print_success(f"Found {len(installed_packages)} installed packages")
63
64 # Get new packages only
65 fmt.print_info("Checking for new packages...")
66 logger.debug(f"Comparing packages with cached list for {machine}")
67 new_packages = pkg_finder.get_new_packages(machine, installed_packages)
68 logger.info(f"Found {len(new_packages)} new packages on {machine}")
69
70 # Update cache with current packages
71 pkg_cache_file = os.path.join(machine_dir, "installed_packages.json")
72 with open(pkg_cache_file, "w") as pkg_file:
73 pkg_file.write(json.dumps(installed_packages))
74 logger.debug(f"Updated package cache for {machine}")
75
76 if not new_packages:
77 fmt.print_warning(f"No new packages to check. Skipping vulnerability analysis.")
78 logger.info(f"No new packages on {machine}. Skipping CPE generation and vulnerability check.")
79 return installed_packages, []
80
81 fmt.print_success(f"Found {len(new_packages)} new packages to analyze")
82 return installed_packages, new_packages

85def generate_cpes_for_packages(new_packages, machine, cpe_matcher):
86 """
87 Generate Common Platform Enumeration (CPE) identifiers for packages.
88
89 Uses Google GenAI API to intelligently map package names to CPE identifiers.
90 Retrieves cached CPEs for previously processed packages to avoid redundant
91 AI model calls.
92
93 @param new_packages list Package names to generate CPEs for
94 @param machine str Machine name for logging and file operations
95 @param cpe_matcher module CPE matcher module with ask_for_cpe() function
96
97 @return dict Dictionary mapping package names to their CPE lists
98 Format: {package_name: [cpe_string_1, cpe_string_2, ...]}
99
100 @details
101 - Filters packages to only process those without cached CPEs
102 - Calls AI model once with all packages (batch operation)
103 - Parses AI response to extract valid CPE strings
104 - Pairs CPEs 1:1 with packages
105 - Caches generated CPEs for future runs
106 - Retrieves previously cached CPEs for packages that have them
107 """
108 fmt.print_info("Generating CPEs for new packages...")
109 logger.info(f"Starting CPE generation for {machine}")
110
111 packages_cpes = {}
112 packages_to_process = pkg_finder.get_packages_needing_cpe_generation(new_packages)
113 logger.debug(f"Packages needing CPE generation: {len(packages_to_process)}")
114
115 if not packages_to_process:
116 fmt.print_info("All packages already have cached CPEs")
117 logger.info(f"All {len(new_packages)} packages on {machine} have cached CPEs")
118 else:
119 model = "gemini-2.5-flash"
120 fmt.print_info(f"Generating CPEs for {len(packages_to_process)} packages (calling AI model: {model})...")
121 logger.info(f"Calling AI model ({model}) to generate CPEs for {len(packages_to_process)} packages on {machine}")
122
123 # Call ask_for_cpe ONCE with all packages needing generation
124 cpe_response = cpe_matcher.ask_for_cpe(packages_to_process, machine, model=model)
125
126 # Parse the response to extract CPEs for each package
127 cpes_from_response = [line.strip() for line in cpe_response.split('\n') if line.strip().startswith('cpe:')]
128 logger.debug(f"Extracted {len(cpes_from_response)} CPEs from AI response")
129
130 # Build a mapping of packages to their CPEs (with validation)
131 valid_cpe_count = 0
132 invalid_cpe_count = 0
133 if cpes_from_response:
134 for i, package in enumerate(packages_to_process):
135 if i < len(cpes_from_response):
136 cpe = cpes_from_response[i].strip()
137 # Validate CPE format before storing
138 if cpe_matcher.validate_cpe_format(cpe):
139 packages_cpes[package] = [cpe]
140 valid_cpe_count += 1
141 logger.debug(f"Valid CPE for {package}: {cpe}")
142 else:
143 logger.warning(f"Invalid CPE format for {package}: {cpe}")
144 invalid_cpe_count += 1
145 packages_cpes[package] = []
146 else:
147 packages_cpes[package] = []
148
149 if invalid_cpe_count > 0:
150 fmt.print_warning(f"Warning: {invalid_cpe_count} invalid CPE formats were rejected")
151 logger.warning(f"Rejected {invalid_cpe_count} invalid CPEs for {machine}")
152
153 # Cache all generated CPEs at once
154 if packages_cpes:
155 pkg_finder.cache_cpes(packages_cpes)
156 fmt.print_success(f"CPEs generated and cached for {len(packages_cpes)} packages ({valid_cpe_count} valid)")
157 logger.info(f"CPEs generated and cached for {len(packages_cpes)} packages on {machine} ({valid_cpe_count} valid, {invalid_cpe_count} invalid)")
158
159 # Add cached CPEs for packages that were already cached
160 for package in new_packages:
161 if package not in packages_cpes:
162 cached_cpes = pkg_finder.get_cached_cpes(package)
163 if cached_cpes:
164 packages_cpes[package] = cached_cpes
165
166 return packages_cpes

169def process_machine_hardware(config, machine):
170 """
171 Retrieve and process hardware information for a machine.
172
173 Connects to a machine using lscpu and extracts CPU/hardware details
174 for CPE generation and vulnerability matching.
175
176 @param config ConfigParser object containing machine configuration
177 @param machine str Machine name from inventory
178
179 @return dict Hardware information dictionary with keys:
180 - vendor_id: CPU vendor (Intel, AMD, etc.)
181 - model_name: CPU model description
182 - family: CPU family identifier
183 - model: CPU model number
184 - stepping: CPU stepping number
185 - flags: CPU flags (spectre, meltdown mitigation, etc.)
186 - cores: Number of CPU cores
187 - threads: Threads per core
188 - raw_output: Complete lscpu output
189
190 @details
191 Used to detect vulnerabilities like:
192 - Spectre (CVE-2017-5753, CVE-2017-5715)
193 - Meltdown (CVE-2017-5754)
194 - CPU errata and microcode issues
195 """
196 if config[machine]['type'] != 'linux':
197 return {}
198
199 fmt.print_info("Retrieving hardware information...")
200 logger.debug(f"Retrieving hardware info from {machine}")
201 hardware_info = pkg_finder.get_hardware_info(config, machine)
202
203 if hardware_info and hardware_info.get('model_name'):
204 fmt.print_success(f"Hardware: {hardware_info.get('model_name', 'Unknown')}")
205 logger.info(f"Retrieved hardware info from {machine}: {hardware_info.get('model_name')}")
206 else:
207 fmt.print_warning("Could not retrieve hardware information")
208 logger.warning(f"No hardware information available from {machine}")
209
210 return hardware_info

213def generate_cpes_for_hardware(hardware_info, machine, cpe_matcher):
214 """
215 Generate CPE identifiers for hardware/CPU information.
216
217 Uses Google GenAI API to map hardware characteristics to CPE identifiers.
218 Similar to package CPE generation but for hardware components.
219
220 @param hardware_info dict Hardware information from lscpu
221 @param machine str Machine name for logging
222 @param cpe_matcher module CPE matcher module with ask_for_cpe() function
223
224 @return dict Dictionary mapping hardware components to CPE lists
225 Example: {'CPU-Intel-Xeon': ['cpe:2.3:h:intel:xeon:...']}
226
227 @details
228 Hardware vulnerabilities are typically microarchitecture-related:
229 - Spectre family (register file/predictor/cache)
230 - Meltdown (TLB)
231 - Side-channel attacks
232 - Bus locking issues
233
234 The CPE matcher identifies the vendor and model to generate appropriate CPEs.
235 """
236 if not hardware_info or not hardware_info.get('model_name'):
237 logger.debug(f"No hardware information to generate CPEs for {machine}")
238 return {}
239
240 fmt.print_info("Generating CPEs for hardware...")
241 logger.info(f"Starting hardware CPE generation for {machine}")
242
243 # Build a simplified hardware description for CPE generation
244 hardware_items = []
245 if hardware_info.get('vendor_id'):
246 hardware_items.append(f"CPU-{hardware_info.get('vendor_id')}")
247
248 if hardware_info.get('model_name'):
249 # Extract main model info, avoid very long descriptions
250 model_parts = hardware_info.get('model_name', '').split(' ')[:3]
251 if model_parts:
252 hardware_items.append("-".join(model_parts))
253
254 if not hardware_items:
255 logger.debug(f"Insufficient hardware information for {machine}")
256 return {}
257
258 model = "gemini-2.5-flash"
259 fmt.print_info(f"Generating CPEs for hardware component (calling AI model: {model})...")
260 logger.info(f"Calling AI model ({model}) to generate hardware CPEs for {machine}")
261
262 # Generate CPE for hardware
263 hardware_cpe_response = cpe_matcher.ask_for_cpe(hardware_items, machine, model=model, is_hardware=True)
264
265 # Parse response
266 hardware_cpes = {}
267 cpes_from_response = [line.strip() for line in hardware_cpe_response.split('\n') if line.strip().startswith('cpe:')]
268 logger.debug(f"Extracted {len(cpes_from_response)} hardware CPEs from AI response")
269
270 if cpes_from_response:
271 for i, hw_item in enumerate(hardware_items):
272 if i < len(cpes_from_response):
273 cpe = cpes_from_response[i].strip()
274 if cpe_matcher.validate_cpe_format(cpe):
275 hardware_cpes[hw_item] = [cpe]
276 logger.debug(f"Valid hardware CPE: {cpe}")
277 else:
278 logger.warning(f"Invalid hardware CPE format: {cpe}")
279
280 if hardware_cpes:
281 fmt.print_success(f"Hardware CPEs generated for {len(hardware_cpes)} components")
282 logger.info(f"Generated hardware CPEs for {len(hardware_cpes)} components on {machine}")
283 else:
284 fmt.print_warning("No valid hardware CPEs generated")
285 logger.warning(f"Failed to generate valid hardware CPEs for {machine}")
286
287 return hardware_cpes
